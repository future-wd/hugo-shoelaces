{{- /* declare variables */ -}} {{- $targetDir := "css" -}} {{- $targetPath := "" -}} {{- $file := "" -}} {{- /* assign variables from dict if present */ -}} {{- if reflect.IsMap . -}} {{- with .targetDir }} {{- $targetDir = .-}} {{- end -}} {{- with .targetPath -}} {{- $targetPath = . -}} {{- end -}} {{- $file = .src -}} {{- else -}} {{- /* if dict not present, assign context to $file */ -}} {{- $file = . }} {{- end -}} {{- /* remove first directory from path if present */ -}} {{- $fileCSS := $file }} {{- if findRE "(.*)/" $fileCSS -}} {{- $fileCSS = replaceRE "(.*)/" "" $fileCSS -}} {{- if strings.HasSuffix $fileCSS (or "scss" "sass") -}} {{ $fileCSS = substr $fileCSS 0 -4 -}} {{ $fileCSS = printf "%s.css" $fileCSS -}} {{- end -}} {{- end -}} {{- if not $targetPath -}} {{- $targetPath = printf "%s/%s" $targetDir $fileCSS -}} {{- end -}} {{- /* parse CSS into libsass compiler */ -}} {{- with $src := resources.Get $file -}} {{ if eq (hugo.Environment) "development" -}} {{ $options := (dict "targetPath" $targetPath "transpiler" "libsass" "enableSourceMap" true "includePaths" (slice "node_modules")) -}} {{ $src = $src | toCSS $options | postCSS (dict "config" "config/postcss.config.js") |resources.Fingerprint -}} {{ else -}} {{ $options := (dict "targetPath" $targetPath "transpiler" "libsass" "outputStyle" "compressed" "includePaths" (slice "node_modules")) -}} {{ $src = $src | toCSS $options | postCSS (dict "config" "config/postcss.config.js") | resources.Fingerprint -}} {{ end -}} {{- end -}}
